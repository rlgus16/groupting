rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // ========== 사용자(users) 컬렉션 ==========
    match /users/{userId} {
      // 사용자 생성: 로그인한 사용자가 본인 문서만 생성 가능
      allow create: if request.auth != null && request.auth.uid == userId;
      
      // 사용자 조회: 로그인한 모든 사용자가 다른 사용자 프로필 조회 가능 (매칭을 위해 필요)
      allow read: if request.auth != null;
      
      // 사용자 수정: 본인 정보만 수정 가능
      allow update: if request.auth != null && request.auth.uid == userId;
      
      // 사용자 삭제: 본인 계정만 삭제 가능
      allow delete: if request.auth != null && request.auth.uid == userId;
    }
    
    // ========== 그룹(groups) 컬렉션 ==========
    match /groups/{groupId} {
      // 그룹 생성: 로그인한 사용자 누구나 그룹 생성 가능
      allow create: if request.auth != null;
      
      // 그룹 조회: 그룹 멤버이거나 매칭 대상 그룹 조회 가능
      allow read: if request.auth != null && (
        // 그룹 멤버인 경우
        request.auth.uid in resource.data.memberIds ||
        // 매칭 중인 그룹은 누구나 조회 가능
        resource.data.status == 'matching'
      );
      
      // 그룹 수정: 그룹 멤버만 수정 가능
      allow update: if request.auth != null && 
        request.auth.uid in resource.data.memberIds;
      
      // 그룹 삭제: 그룹 생성자(방장)만 삭제 가능
      allow delete: if request.auth != null && 
        request.auth.uid == resource.data.ownerId;
    }
    
    // ========== 메시지(messages) 컬렉션 ==========
    match /messages/{messageId} {
      // 메시지 생성: 본인이 보내는 메시지이고, 해당 그룹의 멤버인 경우만 가능
      allow create: if request.auth != null && 
        request.auth.uid == request.resource.data.senderId &&
        isGroupMember(request.resource.data.groupId, request.auth.uid);
      
      // 메시지 조회: 해당 그룹의 멤버만 조회 가능
      allow read: if request.auth != null && 
        isGroupMember(resource.data.groupId, request.auth.uid);
      
      // 메시지 수정: 메시지 작성자만 수정 가능 (읽음 처리 등)
      allow update: if request.auth != null && 
        request.auth.uid == resource.data.senderId;
      
      // 메시지 삭제: 메시지 작성자만 삭제 가능 (시스템 메시지는 삭제 불가)
      allow delete: if request.auth != null && 
        request.auth.uid == resource.data.senderId &&
        resource.data.type != 'system';
    }
    
    // ========== 초대장(invitations) 컬렉션 ==========
    match /invitations/{invitationId} {
      // 초대장 생성: 본인이 보내는 초대장만 생성 가능
      allow create: if request.auth != null && 
        request.auth.uid == request.resource.data.fromUserId;
      
      // 초대장 조회: 초대 보낸 사람과 받는 사람만 조회 가능
      allow read: if request.auth != null && (
        request.auth.uid == resource.data.fromUserId ||
        request.auth.uid == resource.data.toUserId
      );
      
      // 초대장 수정: 초대 받은 사람만 수락/거절 상태 변경 가능
      allow update: if request.auth != null && 
        request.auth.uid == resource.data.toUserId;
      
      // 초대장 삭제: 초대 보낸 사람만 삭제 가능
      allow delete: if request.auth != null && 
        request.auth.uid == resource.data.fromUserId;
    }
    
    // ========== 헬퍼 함수들 ==========
    
    // 사용자가 특정 그룹의 멤버인지 확인
    function isGroupMember(groupId, userId) {
      return exists(/databases/$(database)/documents/groups/$(groupId)) &&
        userId in get(/databases/$(database)/documents/groups/$(groupId)).data.memberIds;
    }
    
    // 사용자가 특정 그룹의 방장인지 확인
    function isGroupOwner(groupId, userId) {
      return exists(/databases/$(database)/documents/groups/$(groupId)) &&
        userId == get(/databases/$(database)/documents/groups/$(groupId)).data.ownerId;
    }
    
    // 매칭된 그룹들의 멤버인지 확인 (확장 기능용)
    function isMatchedGroupMember(groupId, userId) {
      let group = get(/databases/$(database)/documents/groups/$(groupId)).data;
      return userId in group.memberIds ||
        (group.matchedGroupId != null && 
         exists(/databases/$(database)/documents/groups/$(group.matchedGroupId)) &&
         userId in get(/databases/$(database)/documents/groups/$(group.matchedGroupId)).data.memberIds);
    }
  }
}
